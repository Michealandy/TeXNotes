161        """

170            elif m.group(1) is not None and m.group(3) is None:
171                return f'\\\\exhyperref{new_reference}'
172            elif m.group(1) is not None:
173                return f'\\\\exhyperref[{m.group(4)}]{{{new_reference}}}'

178        note = database.Note.get(reference=old_reference)
179
180        lines = bytearray()
181
182        with open('notes/documents.tex', 'r') as f:
183            for line in f:
184                lines.extend(re.sub('\\\\externaldocument\[' + note.reference + '\-\]\{' + note.filename + '\}', '\\\\externaldocument[' + new_reference + '-]{' + note.filename + '}', line).encode())

252        """
253
254            Render the LaTeX file. By default renders as a pdf and output is stored in the folder /pdf. The other format option is html, although support is currently experimental.
255
256        """
257        import subprocess
258        command, options = Helper.renderers[format]
259        
260        try:
261            os.mkdir(format)
262        except FileExistsError:
263            pass
264
265        
266        note = database.Note.get(filename=filename)
267        linked_files = set()
268
269        external_documents = ""
270        references = set()
271
272        for label in note.labels:
273            for link in label.referenced_by:
274                source = link.source
275                references.add(source)
276                linked_files.add(link.source.reference)
277
278
279
280
281        for link in note.references:
282            target_note = link.target.note
283            if link.target.note.last_build_date_html is not None:
284                references.add(target_note) 
285      
286        #inject external documents 
287        if format == 'html':
288            for reference in references:
289                if reference.last_build_date_html is not None:
290                    external_documents += f"\\externaldocument[{reference.reference}-]{{{reference.filename}}}\n"
291       
292        referenced_by_section = "\\section*{Referenced In}\n\\begin{itemize}\n"
293        for reference in linked_files:
294            referenced_by_section += f"\\item \\excref{{{reference}}}"
295
296        referenced_by_section += "\\end{itemize}"
297
298        
299
300        os.chdir(format)
301
302
303        path_to_file = f'../notes/slipbox/{filename}.tex'
304
305
306        try:
307            with open(path_to_file, 'r'):
308                pass
309        except FileNotFoundError:
310            print('No such file!')
311            return '', f'Can\'t find {path_to_file}'
312
313
314        with open(path_to_file, 'r') as f:
315            contents = f.read()
316
317        if format == 'pdf':
318            options.insert(0, f"--jobname={filename}")
319        elif format == 'html':
320            options = ['-j', filename] + options + ['"svg"']
321            
322 
323
324        document = contents.split('\\end{document}')[0]
325
326        if len(linked_files) > 0:
327            document += referenced_by_section
328
329        if format == 'html':
330            #replace include with preamble_html and inject external documents
331            document = document.replace("\\subimport{../template}{preamble.tex}", "\\subimport{../template}{preamble_html.tex}\n" + external_documents)
332
333        document += "\\end{document}"
334
335
336        process = subprocess.run([command, *options], input=document.encode(), capture_output=True)
337
338        os.chdir('../')
339
340        if format == 'html':
341            note.last_build_date_html = datetime.datetime.now()
342        elif format == 'pdf':
343            note.last_build_date_pdf = datetime.now()
344        note.save()
345        return process.stdout, process.stderr

654    def export_draft(input_file, output_file=None):
655        """
656            Exports the export input_file (a file like /export/example.tex containing \ExecuteMetaData calls) and creates a .tex file (by default in /draft/filename.tex) where the \ExecuteMetadata commands are replaced with the body of the notes that they reference.
657        """
658        if output_file is None:
659            try:
660                os.mkdir('draft')
661            except FileExistsError:
662                pass
663            filename = input_file.split('/')[-1]
664            output_file = f'draft/{filename}'
665
666        output = bytearray()
667    
668        with open(input_file, 'r') as f:
669            for line in f:
670                output.extend((re.sub('\\\\ExecuteMetaData\[\.\./([^]]+)\]\{([^}]+)\}', '', line).strip() + '\n').encode())
671                external_documents = re.finditer('\\\\ExecuteMetaData\[\.\./([^]]+)\]\{([^}]+)\}', line)
672                for document in external_documents:
673                    import_file = document.group(1) 
674                    tag = document.group(2)
675                    with open(import_file, 'r') as in_file:
676                        import_text_file = in_file.read()
677                        import_text = re.search(f'%<\*{tag}>((.|\n)*?)%</{tag}>', import_text_file)
678                        output.extend(import_text.group(1).strip().encode())
679
680
681        with open(output_file, 'wb') as f:
682            f.write(output)

160        """
161            Rename the reference used throughout the whole Zettelkasten. This function changes documents.tex and also any documents that reference this note.
162        """

166        def replace_text(m):
167            if m.group(1) is None and m.group(3) is None:
168                return f'\\\\excref{new_reference}'
169            elif m.group(1) is None and m.group(3) is not None:
170                return f'\\\\excref[{m.group(4)}]{{{new_reference}}}'
171            elif m.group(1) is not None and m.group(3) is None:
172                return f'\\\\exhyperref{new_reference}'
173            elif m.group(1) is not None:
174                return f'\\\\exhyperref[{m.group(4)}]{{{new_reference}}}'

176
177        # Update documents.tex
178
179        note = database.Note.get(reference=old_reference)
180
181        lines = bytearray()
182

184            for line in f:

253        """
254
255            Render the LaTeX file. By default renders as a pdf and output is stored in the folder /pdf. The other format option is html, although support is currently experimental.
256
257        """
258        import subprocess
259        command, options = Helper.renderers[format]
260        
261        try:
262            os.mkdir(format)
263        except FileExistsError:
264            pass
265
266        
267        note = database.Note.get(filename=filename)
268        linked_files = set()
269
270        external_documents = ""
271        references = set()
272
273        for label in note.labels:
274            for link in label.referenced_by:
275                source = link.source
276                references.add(source)
277                linked_files.add(link.source.reference)
278
279
280
281
282        for link in note.references:
283            target_note = link.target.note
284            if link.target.note.last_build_date_html is not None:
285                references.add(target_note) 
286      
287        #inject external documents 
288        if format == 'html':
289            for reference in references:
290                if reference.last_build_date_html is not None:
291                    external_documents += f"\\externaldocument[{reference.reference}-]{{{reference.filename}}}\n"
292       
293        referenced_by_section = "\\section*{Referenced In}\n\\begin{itemize}\n"
294        for reference in linked_files:
295            referenced_by_section += f"\\item \\excref{{{reference}}}"
296
297        referenced_by_section += "\\end{itemize}"
298
299        
300
301        os.chdir(format)
302
303
304        path_to_file = f'../notes/slipbox/{filename}.tex'
305
306
307        try:
308            with open(path_to_file, 'r'):
309                pass
310        except FileNotFoundError:
311            print('No such file!')
312            return '', f'Can\'t find {path_to_file}'
313
314
315        with open(path_to_file, 'r') as f:
316            contents = f.read()
317
318        if format == 'pdf':
319            options.insert(0, f"--jobname={filename}")
320        elif format == 'html':
321            options = ['-j', filename] + options + ['"svg"']
322            
323 
324
325        document = contents.split('\\end{document}')[0]
326
327        if len(linked_files) > 0:
328            document += referenced_by_section
329
330        if format == 'html':
331            #replace include with preamble_html and inject external documents
332            document = document.replace("\\subimport{../template}{preamble.tex}", "\\subimport{../template}{preamble_html.tex}\n" + external_documents)
333
334        document += "\\end{document}"
335
336
337        process = subprocess.run([command, *options], input=document.encode(), capture_output=True)
338
339        os.chdir('../')
340
341        if format == 'html':
342            note.last_build_date_html = datetime.datetime.now()
343        elif format == 'pdf':
344            note.last_build_date_pdf = datetime.now()
345        note.save()
346        return process.stdout, process.stderr
